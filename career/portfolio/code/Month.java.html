019 package mk.util.calendar;
020 
021 import java.io.StringWriter;
022 import java.text.*;
023 import java.util.*;
024 
025 import org.apache.log4j.Logger;
026 
027 /**
028  * A composition of Week objects.
029  *
030  * @author  murali
031  */
032 public class Month extends Object implements Printable {
033 
034     //------------------------------------------------------------
035     //- Class Variables
036 
037     protected static final Logger LOG = Logger.getLogger(Month.class);
038 
039     protected static final DateFormat HEADER_FORMAT = new SimpleDateFormat("MMMM yyyy");
040 
041     //------------------------------------------------------------
042     //- Class Functions
043 
044 
045     //------------------------------------------------------------
046     //- Instance Variables
047 
048     private final Collection vWeeks = new ArrayList();
049     private String vHeader = "";
050     private Week vCurrentWeek = Week.UNDEFINED;
051 
052 
053     //------------------------------------------------------------
054     //- Constructors
055 
056     public Month(DayDate pDate) {
057         getWeeks().add(Week.HEADER);
058 
059         Iterator iterator = pDate.toAllDaysOfMonth().iterator();
060         while (iterator.hasNext()) {
061             DayDate day = (DayDate) iterator.next();
062             add(day);
063         }
064 
065         setCurrentWeek(Week.UNDEFINED);
066     }
067 
068 
069     //------------------------------------------------------------
070     //- Accessors
071 
072     public Collection getWeeks() {
073         return (vWeeks);
074     }
075 
076     protected String getHeader() {
077         return (vHeader);
078     }
079 
080     private Week getCurrentWeek() {
081         return (vCurrentWeek);
082     }
083 
084 
085     //------------------------------------------------------------
086     //- Settors
087 
088     protected void setHeader(String pHeader) {
089         vHeader = pHeader;
090     }
091 
092     private void setCurrentWeek(Week pCurrentWeek) {
093         vCurrentWeek = pCurrentWeek;
094     }
095 
096 
097     //------------------------------------------------------------
098     //- Private/Protected Utility Functions
099 
100     protected void add(DayDate pDayDate) {
101         DayOfMonth dom = new DayOfMonth(pDayDate);
102 
103         if ("".equals(getHeader())) {
104             setHeader(pDayDate.format(HEADER_FORMAT));
105         }
106 
107         if (!getCurrentWeek().isDefined()) {
108             Week next = new Week();
109             setCurrentWeek(next);
110             getWeeks().add(next);
111         }
112 
113         if (getCurrentWeek().add(dom)) {
114             setCurrentWeek(Week.UNDEFINED);
115         }
116     }
117 
118 
119     public void mark(Range pRange) {
120         Iterator iterator = getWeeks().iterator();
121         while (iterator.hasNext()) {
122             Week week = (Week) iterator.next();
123             week.mark(pRange);
124         }
125     }
126 
127 
128     //------------------------------------------------------------
129     //- Public Interface Functions
130 
131     public String toString() {
132         StringBuffer buffer = new StringBuffer();
133 
134         int padSize = (20 - getHeader().length()) / 2;
135         for (int i = 0; i < padSize; i++) {
136             buffer.append(' ');
137         }
138         buffer.append(getHeader()).append('\n');
139 
140         Iterator iterator = getWeeks().iterator();
141         while (iterator.hasNext()) {
142             Week week = (Week) iterator.next();
143             buffer.append(week).append('\n');
144         }
145 
146         return (buffer.toString());
147     }
148 
149 
150     public void toHtml(StringWriter pWriter) {
151         pWriter.write("<table cellspacing=\"0\">\n");
152 
153         pWriter.write("<tr><th colspan=\"7\" bgcolor=\"" + HEADER_COLOR + "\">");
154         pWriter.write(getHeader());
155         pWriter.write("</th></tr>\n");
156 
157         Iterator iterator = getWeeks().iterator();
158         while (iterator.hasNext()) {
159             Week week = (Week) iterator.next();
160             week.toHtml(pWriter);
161         }
162 
163         pWriter.write("</table>\n");
164     }
