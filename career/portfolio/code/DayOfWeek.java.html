026 /**
027  * An enumeration of the representation of the day of the week.
028  * The underlying id is the same value used in Calendar.
029  *
030  * @author  murali
031  */

035     //- Class Variables
036 
037     protected static final Logger LOG = Logger.getLogger(DayOfWeek.class);
038 
039     private static final int UNDEFINED_ID = -1;
040 
041     // Note: this could become corrupted if public.
042     private static final List ALL = new ArrayList();
043 
044     // Note: this could become corrupted if public.
045     private static final Map MAP = new HashMap();
046 
047     public static final DayOfWeek SUN = new Sunday();
048     public static final DayOfWeek MON = new Monday();
049     public static final DayOfWeek TUE = new Tuesday();
050     public static final DayOfWeek WED = new Wednesday();
051     public static final DayOfWeek THU = new Thursday();
052     public static final DayOfWeek FRI = new Friday();
053     public static final DayOfWeek SAT = new Saturday();
054     public static final DayOfWeek UNDEFINED = new Undefined();
055 
056

058     //- Class Functions
059 
060     public static DayOfWeek get(int pId) {
061         Integer key = new Integer(pId);
062         DayOfWeek candidate = (DayOfWeek) MAP.get(key);
063         DayOfWeek result = (candidate == null) ? UNDEFINED : candidate;
064 
065         return (result);
066     }
067 
068     public static DayOfWeek get(Calendar pCalendar) {
069         int id = pCalendar.get(Calendar.DAY_OF_WEEK);
070         DayOfWeek result = get(id);
071         return (result);
072     }
073 
074 
075     public static Iterator iterator() {
076         Iterator result = ALL.iterator();
077         return (result);
078     }

084     private final int vId;
085 
086     //------------------------------------------------------------
087     //- Constructors
088 
089     protected DayOfWeek(int pId) {
090         vId = pId;
091 
092         if (UNDEFINED_ID != pId) {
093             ALL.add(this);
094             MAP.put(new Integer(pId), this);
095         }
096     }

122     public boolean equals(Object object) {
123         boolean result = false;
124 
125         if (object instanceof DayOfWeek) {
126             DayOfWeek other = (DayOfWeek) object;
127             result = (getId() == other.getId());
128         }
129 
130         return (result);
131     }

134     public int hashCode() {
135         int result = getId();
136         return (result);
137     }
138 
139 
140     public int compareTo(Object object) {
141         int result = 0;
142 
143         if (object instanceof DayOfWeek) {
144             DayOfWeek other = (DayOfWeek) object;
145             Integer idThis = new Integer(getId());
146             Integer idOther = new Integer(other.getId());
147             result = idThis.compareTo(idOther);
148         }
149 
150         return (result);
151     }
