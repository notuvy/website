019 package mk.util.calendar;
020 
021 import java.text.*;
022 import java.util.*;
023 
024 import org.apache.log4j.Logger;
025 
026 /**
027  * The object representing the actual day as a time object.
028  * We could not simply use <code>Date</code> or <code>Calendar</code>
029  * because neither give sufficient functionality in a single object.
030  * This does by combining features from both.
031  *
032  * All the date/time manipulation logic is isolated to this class.
033  *
034  * @author  murali
035  */
036 public class DayDate extends Object implements Comparable {
037 
038     //------------------------------------------------------------
039     //- Class Variables
040 
041     protected static final Logger LOG = Logger.getLogger(DayDate.class);
042 
043     /**
044      * The simple input format for any day of the year.
045      */
046     protected static final DateFormat LONG_FORMAT = new SimpleDateFormat("yyyy-MM-dd");
047 
048 
049     private static final Date UNDEFINED_DATE = new Date(0);
050     public static final DayDate UNDEFINED = new DayDate(0);
051 
052 
053     //------------------------------------------------------------
054     //- Class Functions
055 
056     protected static void truncate(Calendar pCalendar) {
057         pCalendar.set(Calendar.HOUR, 0);
058         pCalendar.set(Calendar.MINUTE, 0);
059         pCalendar.set(Calendar.SECOND, 0);
060         pCalendar.set(Calendar.MILLISECOND, 0);
061     }
062 
063 
064     //------------------------------------------------------------
065     //- Instance Variables
066 
067     private final Date vDate;
068 
069 
070     //------------------------------------------------------------
071     //- Constructors
072 
073     private DayDate(long pLong) {
074         vDate = new Date(pLong);
075     }
076 
077     public DayDate(Date pDate) {
078         Calendar calendar = Calendar.getInstance();
079         calendar.setTime(pDate);
080         truncate(calendar);
081         vDate = calendar.getTime();
082     }
083 
084     public DayDate(Calendar pCalendar) {
085         truncate(pCalendar);
086         vDate = pCalendar.getTime();
087     }
088 
089     public DayDate(String pString) {
090         Date candidate = UNDEFINED_DATE;  // Fallback
091         try {
092             candidate = LONG_FORMAT.parse(pString);
093         } catch (ParseException pe) {
094             LOG.error("Could not parse [" + pString + "].", pe);
095         }
096 
097         if (candidate != UNDEFINED_DATE) {
098             Calendar calendar = Calendar.getInstance();
099             calendar.setTime(candidate);
100             truncate(calendar);
101             candidate = calendar.getTime();
102         }
103 
104         vDate = candidate;
105     }
106 
107 
108     public DayDate() {
109         this (new Date());
110     }
111 
112 
113     //------------------------------------------------------------
114     //- Accessors
115 
116     public Date getDate() {
117         return (vDate);
118     }
119 
120 
121     //------------------------------------------------------------
122     //- Settors
123 
124 
125     //------------------------------------------------------------
126     //- Private/Protected Utility Functions
127 
128     protected Calendar toFirstCalendar() {
129         Calendar result = toCalendar();
130         result.set(Calendar.DAY_OF_MONTH, 1);
131 
132         return (result);
133     }
134 
135     protected DayDate toFirst() {
136         DayDate result = new DayDate(toFirstCalendar());
137         return (result);
138     }
139 
140 
141     //------------------------------------------------------------
142     //- Public Interface Functions
143 
144     public String format(DateFormat pFormat) {
145         String result = pFormat.format(getDate());
146         return (result);
147     }
148 
149 
150     public String toString() {
151         String result = format(LONG_FORMAT);
152         return (result);
153     }
154 
155 
156     public Calendar toCalendar() {
157         Calendar result = Calendar.getInstance();
158         result.setTime(getDate());
159 
160         return (result);
161     }
162 
163 
164     public DayOfWeek toDayOfWeek() {
165         DayOfWeek result = isDefined() ?
166             DayOfWeek.get(toCalendar()) :
167             DayOfWeek.UNDEFINED;
168 
169         return (result);
170     }
171 
172 
173     public Collection toAllDaysOfMonth() {
174         Collection result = new ArrayList();
175 
176         if (isDefined()) {
177             Calendar cal = toFirstCalendar();
178             int thisMonth = cal.get(Calendar.MONTH);
179             do {
180                 result.add(new DayDate(cal));
181                 cal.add(Calendar.DAY_OF_MONTH, 1);
182             } while (cal.get(Calendar.MONTH) == thisMonth);
183         }
184 
185         return (result);
186     }
187 
188 
189     public DayDate nextDay() {
190         Calendar cal = toCalendar();
191         cal.add(Calendar.DAY_OF_MONTH, 1);
192         DayDate result = new DayDate(cal);
193 
194         return (result);
195     }
196 
197 
198     public Collection enumerateTo(DayDate pEnd) {
199         Collection result = new ArrayList();
200 
201         for (DayDate day=this; !day.after(pEnd); day=day.nextDay()) {
202             result.add(day);
203         }
204 
205         return (result);
206     }
207 
208 
209     public boolean isDefined() {
210         boolean result = !UNDEFINED_DATE.equals(getDate());
211         return (result);
212     }
213 
214 
215     public boolean before(DayDate pOther) {
216         boolean result = (pOther != null) && getDate().before(pOther.getDate());
217         return (result);
218     }
219 
220 
221     public boolean after(DayDate pOther) {
222         boolean result = (pOther != null) && getDate().after(pOther.getDate());
223         return (result);
224     }
225 
226 
227     //------------------------------------------------------------
228     //- Class Interface Functions
229 
230     public boolean equals(Object object) {
231         boolean result = false;  // Pessimistic.
232 
233         if (object instanceof DayDate) {
234             DayDate other = (DayDate) object;
235             result = getDate().equals(other.getDate());
236         }
237 
238         return (result);
239     }
240 
241 
242     public int hashCode() {
243         int result = getDate().hashCode();
244         return (result);
245     }
246 
247 
248     public int compareTo(Object object) {
249         int result = 0;
250 
251         if (object instanceof DayDate) {
252             DayDate other = (DayDate) object;
253             result = getDate().compareTo(other.getDate());
254         }
255 
256         return (result);
257     }
