021 import java.util.*;
022 
023 import org.apache.log4j.Logger;

025 /**
026  * Represent a contiguous range of dates.
027  * Implemented by just storing the start and end dates of the range.
028  * Any number of additional dates can be added.
029  * If a new date is within the range, then the addition is a noop.
030  * If a new date is outside the range, then the range is expanded.
031  *
032  * @author  murali
033  */
034 public class Range extends Object {
035 
036     //------------------------------------------------------------
037     //- Class Variables
038 
039     protected static final Logger LOG = Logger.getLogger(Range.class);
040 
041 
042     //------------------------------------------------------------
043     //- Class Functions
044 
045 
046     //------------------------------------------------------------
047     //- Instance Variables
048 
049     private DayDate vStart = DayDate.UNDEFINED;
050     private DayDate vEnd = DayDate.UNDEFINED;
051 
052 
053     //------------------------------------------------------------
054     //- Constructors
055 
056     public Range(DayDate pStart, DayDate pEnd) {
057         vStart = pStart;
058         vEnd = pEnd;
059     }
060 
061     public Range(String pStart, String pEnd) {
062         this (new DayDate(pStart), new DayDate(pEnd));
063     }
064 
065 
066     public Range(String pSingle) {
067         this (new DayDate(pSingle), new DayDate(pSingle));
068     }
069 
070 
071     public Range() {
072     }
073 
074 
075     //------------------------------------------------------------
076     //- Accessors
077 
078     public DayDate getStart() {
079         return (vStart);
080     }
081 
082     public DayDate getEnd() {
083         return (vEnd);
084     }
085 
086 
087     //------------------------------------------------------------
088     //- Settors
089 
090     public void setStart(DayDate pStart) {
091         vStart = pStart;
092     }
093 
094     public void setEnd(DayDate pEnd) {
095         vEnd = pEnd;
096     }
097 
098 
099     //------------------------------------------------------------
100     //- Private/Protected Utility Functions
101 
102 
103     //------------------------------------------------------------
104     //- Public Interface Functions
105 
106     public boolean isDefined() {
107         boolean result = getStart().isDefined() && getEnd().isDefined();
108         return (result);
109     }
110 
111 
112     public void add(DayDate pDate) {
113         if (pDate.isDefined()) {
114             if (!getStart().isDefined() || pDate.before(getStart())) {
115                 setStart(pDate);
116             }
117 
118             if (!getEnd().isDefined() || pDate.after(getEnd())) {
119                 setEnd(pDate);
120             }
121         }
122     }
123 
124 
125     /**
126      * If this instance is undefined, then define it to be for
127      * only today.
128      */
129     public void sanitize() {
130         if (!isDefined()) {
131             add(new DayDate());
132         }
133     }
134 
135 
136     public String toString() {
137         String result = isDefined() ?
138             "[" + getStart() + ", " + getEnd() + "]" :
139             "[Undefined Range]";
140         return (result);
141     }
142 
143 
144     public Collection toMonths() {
145         Collection result = new ArrayList();
146 
147         if (isDefined()) {
148             SortedSet days = new TreeSet();
149             Collection alldays = getStart().enumerateTo(getEnd());
150             Iterator iterator = alldays.iterator();
151             while (iterator.hasNext()) {
152                 DayDate day = (DayDate) iterator.next();
153                 DayDate first = day.toFirst();
154                 days.add(first);
155             }
156 
157             iterator = days.iterator();
158             while (iterator.hasNext()) {
159                 DayDate day = (DayDate) iterator.next();
160                 Month month = new Month(day);
161                 month.mark(this);
162                 result.add(month);
163             }
164         }
165 
166         return (result);
167     }
168 
169 
170     public boolean contains(DayDate pDayDate) {
171         boolean isBefore = pDayDate.before(getStart());
172         boolean isAfter = pDayDate.after(getEnd());
173         boolean outside = isBefore || isAfter;
174         boolean result = !outside;
175         return (result);
176     }
177 
178 
179     //------------------------------------------------------------
180     //- Class Interface Functions
181 
182     public boolean equals(Object object) {
183         boolean result = false;  // Pessimistic.
184 
185         if (object instanceof Range) {
186             Range other = (Range) object;
187             result = (getStart().equals(other.getStart()) &&
188                       getEnd().equals(other.getEnd()));
189         }
190 
191         return (result);
192     }
